06. 배열 : 데이터 불러오기 : for ( )문

{
    const arr = [100, 200, 300, 400, 500];

    document.write(arr[0]);
    document.write(arr[1]);
    document.write(arr[2]);
    document.write(arr[3]);
    document.write(arr[4]);
    
    //for(초기값,조건식,증감식)
    for(let i=0; i<arr.length; i++){
        document.write(arr[i], "<br>");
}

결과보기

100
200
300
400
500

07. 배열 : 데이터 불러오기 : forEach

{
    const num = [100, 200, 300, 400, 500];

    document.write(num[0])
    document.write(num[1])
    document.write(num[2])
    document.write(num[3])
    document.write(num[4])
    //for문
    for( let i=0; i<num.length; i++ ){
        document.write(num[i],"<br>");
    };
    //forEach()
    num.forEach(function(el){
        document.write(el, "<br>");
    });
    //forEach(element, index, array)// 매개변수 element, index, array: 요소, 인덱스(순서), 어레이(배열)
    num.forEach(function(element, index, array){
        document.write(element);
        document.write(index);
        document.write(array);
    });
}

결과보기

100 200 300 400 500
100 200 300 400 500
100 0 100 200 300 400 500
200 1 100 200 300 400 500
300 2 100 200 300 400 500
400 3 100 200 300 400 500
500 4 100 200 300 400 500

08. 배열 : 데이터 불러오기 : for of

{
    const num = [100, 200, 300, 400, 500];
    
    for( let i of num ){
        document.write(i); // for of는 배열에서 쓰기 편한 방법.
    };
}

결과보기

100 200 300 400 500

09. 배열 : 데이터 불러오기 : for in

{
    const num = [100, 200, 300, 400, 500];
    
    for( let i in num ){
        document.write(i); // index를 불러온다. for in은 객체에서 쓰기 편한 방법.
    };
    for( let i in num ){
        document.write(num[i]); //for of 처럼 불러온다.
    };
}

결과보기

0 1 2 3 4
100 200 300 400 500

11. 배열 : 데이터 불러오기 : 펼침 연산자(Spread Operator)

{
    const num = [100, 200, 300, 400, 500];

    document.write(num); //100,200,300,400,500 (쉼표가 같이 출력: 하나의 데이터로 봄)

    document.write(num[0], num[1], num[2], num[3], num[4]); //쉼표 없이 각자의 데이터로 출력

    document.write(...num); //100200300400500       
}

결과보기

100, 200, 300, 400, 500
100 200 300 400 500
100 200 300 400 500

14. 객체 : 데이터 불러오기 : Object

{
    const obj = {
        a: 100,
        b: 200,
        c: "javascript"
    };
    
    document.write(Object.keys(obj)); //키 불러오기
    document.write(Object.values(obj)); // 밸류 불러오기
    document.write(Object.entries(obj)); // 키와 밸류 다 불러오기        
}

결과보기

a, b, c
100, 200, javascript
a, 100, b, 200, c, javascript
