1. 1부터 100까지 출력하기

1.1 while문을 이용해서 1부터 100까지 출력하는 함수를 작성해주세요.

{
    function func( ){
        let num = 1;
    
        while(num<=100){
            console.log(num);
            num++;
        }
    }
    func( );
}

1.2 for문을 이용해서 1부터 100까지 출력하는 함수를 작성해주세요.

{
    function func( ){
        for(let i =1; i<=100; i++){
            console.log(i);
        }   
    }
    func( );
}

1.2 for문을 이용해서 1부터 100까지 출력하는 함수를 작성해주세요.

조건1. 짝수만 출력하세요!

조건2. 홀수만 출력하세요!

짝수

{
    function func( ){
        for(let i =1; i<=100; i++){
            if(i%2==0){
                console.log(i);
            }
        }
    }
    func( );    
    //다른방법1 (if 괄호 생략)
    for(let i =1; i<=100; i++){
        if(i%2==0)
            console.log(i);
    }
    //다른 방법2
    for(let i =2; i<=100; i+=2){        
            console.log(i);
    }
}

홀수

{
    function func( ){
